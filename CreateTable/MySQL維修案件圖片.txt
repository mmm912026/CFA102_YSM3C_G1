CREATE TABLE `MAINTENANCE_CASE_IMAGES` (
  `MCI_NO` int NOT NULL AUTO_INCREMENT COMMENT '維修案件圖片編號',
  `MCA_NO` int NOT NULL COMMENT '維修案件編號',
  `MCI_BEFORE_IMG` blob COMMENT '維修前圖片',
  `MCI_AFTER_IMG` blob COMMENT '維修後圖片',
  PRIMARY KEY (`MCI_NO`),
  KEY `FK_MAINTENANCE_MCA_NO` (`MCA_NO`),
  CONSTRAINT `FK_MAINTENANCE_MCA_NO` FOREIGN KEY (`MCA_NO`) REFERENCES `MAINTENANCE_CASE` (`MCA_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='維修案件圖片';
