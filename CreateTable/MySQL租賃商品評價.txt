CREATE TABLE `PRODUCT_REVIEWS` (
  `PR_NO` int NOT NULL AUTO_INCREMENT COMMENT '租賃商品評價編號',
  `RC_NO` int NOT NULL COMMENT '租賃商品類別編號',
  `RO_NO` int NOT NULL COMMENT '租賃訂單編號',
  `PR_CONTENT` varchar(1000) DEFAULT NULL COMMENT '評價內容',
  `PR_IMAGES` longblob COMMENT '租賃圖片',
  `PR_SCORE` int DEFAULT NULL COMMENT '評價分數',
  `PR_STATUS` varchar(20) DEFAULT NULL COMMENT '評價狀態',
  PRIMARY KEY (`PR_NO`),
  KEY `FK_PRODUCT_REVIEWS_RC_NO_idx` (`RC_NO`),
  KEY `FK_PRODUCT_REVIEWS_RO_NO_idx` (`RO_NO`),
  CONSTRAINT `FK_PRODUCT_REVIEWS_RC_NO` FOREIGN KEY (`RC_NO`) REFERENCES `RENTAL_CLASS` (`RC_NO`),
  CONSTRAINT `FK_PRODUCT_REVIEWS_RO_NO` FOREIGN KEY (`RO_NO`) REFERENCES `RENTAL_ORDER` (`RO_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='租賃商品評價';
