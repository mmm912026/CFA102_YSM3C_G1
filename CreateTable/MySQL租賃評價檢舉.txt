CREATE TABLE `REPORT_PRODUCT_REVIEWS` (
  `REP_NO` int NOT NULL AUTO_INCREMENT COMMENT '租賃評價檢舉',
  `PR_NO` int NOT NULL COMMENT '評價編號',
  `MEM_NO` int NOT NULL COMMENT '會員編號',
  `REPORT_CONTENT` varchar(100) DEFAULT NULL COMMENT '檢舉內容',
  `REP_STATUS` varchar(20) DEFAULT NULL COMMENT '檢舉處理狀態',
  `REP_DATE` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '檢舉日期',
  PRIMARY KEY (`REP_NO`),
  KEY `FK_REPORT_PRODUCT_REVIEWS_PR_NO_idx` (`PR_NO`),
  KEY `FK_REPORT_PRODUCT_REVIEWS_MEM_NO_idx` (`MEM_NO`),
  CONSTRAINT `FK_REPORT_PRODUCT_REVIEWS_MEM_NO` FOREIGN KEY (`MEM_NO`) REFERENCES `MEMBER` (`MEM_NO`),
  CONSTRAINT `FK_REPORT_PRODUCT_REVIEWS_PR_NO` FOREIGN KEY (`PR_NO`) REFERENCES `PRODUCT_REVIEWS` (`PR_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='租賃評價檢舉';
