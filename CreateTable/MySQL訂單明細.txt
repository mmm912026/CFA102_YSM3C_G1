CREATE TABLE `SEC_ORDER_LIST` (
  `SOL_NO` int NOT NULL AUTO_INCREMENT COMMENT '二手商品明細編號',
  `SO_NO` int NOT NULL COMMENT '二手商品訂單編號',
  `SPI_NO` int NOT NULL COMMENT '二手商品編號',
  `SOL_PROAMOT` int DEFAULT NULL COMMENT '商品數量',
  `SOL_PRI` int DEFAULT NULL,
  PRIMARY KEY (`SOL_NO`),
  KEY `FK_SEC_ORDER_LIST_SO_NO_idx` (`SO_NO`),
  KEY `FK_SEC_ORDER_LIST_SPI_NO_idx` (`SPI_NO`),
  CONSTRAINT `FK_SEC_ORDER_LIST_SO_NO` FOREIGN KEY (`SO_NO`) REFERENCES `SEC_ORDER` (`SO_NO`),
  CONSTRAINT `FK_SEC_ORDER_LIST_SPI_NO` FOREIGN KEY (`SPI_NO`) REFERENCES `SEC_PRODUCT_INFORMATION` (`SPI_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='訂單明細';
