CREATE TABLE `RENTAL_CLASS` (
  `RC_NO` int NOT NULL AUTO_INCREMENT COMMENT '租賃商品類別編號',
  `RC_NAME` varchar(100) NOT NULL COMMENT '租賃商品類別名稱',
  `RC_ITEM` varchar(100) DEFAULT NULL COMMENT '租賃商品類別',
  `RC_DETAIL` varchar(100) DEFAULT NULL COMMENT '租賃商品類別內容',
  `RC_DEPOSIT` int DEFAULT NULL COMMENT '押金',
  `RC_PRICE` int DEFAULT NULL COMMENT '租賃價格/每天',
  `RC_TOTAL_COUNT` int DEFAULT '0' COMMENT '評價總人數',
  `RC_TOTAL_SCORE` int DEFAULT '0' COMMENT '評價總分數',
  `RC_STORAGE` int DEFAULT NULL COMMENT '庫存',
  `RC_STATUS` varchar(20) DEFAULT NULL COMMENT '租賃商品類別狀態',
  PRIMARY KEY (`RC_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='租賃商品類別';
