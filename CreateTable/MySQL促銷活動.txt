CREATE TABLE `PROMOTIONS` (
  `PROMOTIONS_NO` int NOT NULL AUTO_INCREMENT COMMENT '促銷編號',
  `PROMOTIONS_NAME` char(100) DEFAULT NULL COMMENT '促銷名稱',
  `PROMOTIONS_START_DATE` datetime DEFAULT NULL COMMENT '促銷開始日期',
  `PROMOTIONS_END_DATE` datetime DEFAULT NULL COMMENT '促銷結束日期',
  PRIMARY KEY (`PROMOTIONS_NO`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='促銷活動';
