CREATE TABLE `CONSULTATION` (
  `CONSULT_NO`           int NOT NULL AUTO_INCREMENT COMMENT '諮詢單編號',
  `CONSULTANT`           char(20) DEFAULT NULL COMMENT '諮詢人姓名',
  `CONSULT_PHONE`        varchar(20) DEFAULT NULL COMMENT '諮詢人手機',
  `CONSULT_EMAIL`        varchar(65) DEFAULT NULL COMMENT '諮詢人Email',
  `CONSULT_CONTENT`      varchar(100) DEFAULT NULL COMMENT '諮詢內容',
  `STAFF_NO`             int NOT NULL COMMENT '員工編號',
  `CONSULT_STA`          varchar(20) DEFAULT NULL COMMENT '回覆狀態',
  PRIMARY KEY (`CONSULT_NO`),
  KEY `FK_CONSULT_STAFF_NO` (`STAFF_NO`),
  CONSTRAINT `FK_CONSULT_STAFF_NO` FOREIGN KEY (`STAFF_NO`) REFERENCES `STAFF` (`STAFF_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='諮詢表單';
