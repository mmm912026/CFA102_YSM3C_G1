CREATE TABLE `PERSON_MESSAGE` (
  `PM_NO` int NOT NULL AUTO_INCREMENT COMMENT '訊息編號',
  `MEM_NO` int NOT NULL COMMENT '會員編號',
  `PM_DATE` datetime DEFAULT NULL COMMENT '發送日期',
  `PM_CONTENT` varchar(100) DEFAULT NULL COMMENT '訊息內容',
  `PM_STATUS` varchar(20) DEFAULT NULL COMMENT '訊息狀態',
  PRIMARY KEY (`PM_NO`),
  KEY `PERSON_MESSAGE_MEM_NO_idx` (`MEM_NO`),
  CONSTRAINT `PERSON_MESSAGE_MEM_NO` FOREIGN KEY (`MEM_NO`) REFERENCES `MEMBER` (`MEM_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='訊息總覽';
