CREATE TABLE `SEC_PRODUCT_IMAGES` (
  `SPIM_NO` int NOT NULL AUTO_INCREMENT COMMENT '二手商品圖片編號',
  `SPI_NO` int NOT NULL COMMENT '二手商品編號',
  `SPIM_IMG` mediumblob COMMENT '二手圖片',
  PRIMARY KEY (`SPIM_NO`),
  KEY `FK_SEC_PRODUCT_IMAGES_SPI_NO_idx` (`SPI_NO`),
  CONSTRAINT `FK_SEC_PRODUCT_IMAGES_SPI_NO` FOREIGN KEY (`SPI_NO`) REFERENCES `SEC_PRODUCT_INFORMATION` (`SPI_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='二手商品圖片';
