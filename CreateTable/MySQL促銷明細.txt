CREATE TABLE `PROMOTION_LIST` (
  `PROMOTIONS_NO` int NOT NULL COMMENT '促銷編號',
  `SPI_NO` int NOT NULL COMMENT '產品編號',
  `SPI_PRICE` int DEFAULT NULL COMMENT '產品促銷價格',
  PRIMARY KEY (`PROMOTIONS_NO`,`SPI_NO`),
  KEY `FK_PROMOTION_LIST_PROMOTIONS_NO_idx` (`PROMOTIONS_NO`),
  KEY `FK_PROMOTION_LIST_SPI_NO_idx` (`SPI_NO`),
  CONSTRAINT `FK_PROMOTION_LIST_PROMOTIONS_NO` FOREIGN KEY (`PROMOTIONS_NO`) REFERENCES `PROMOTIONS` (`PROMOTIONS_NO`),
  CONSTRAINT `FK_PROMOTION_LIST_SPI_NO` FOREIGN KEY (`SPI_NO`) REFERENCES `SEC_PRODUCT_INFORMATION` (`SPI_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='促銷明細';
