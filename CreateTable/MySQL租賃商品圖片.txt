CREATE TABLE `RENTAL_PRODUCT_IMAGES` (
  `RPI_NO` int NOT NULL AUTO_INCREMENT COMMENT '租賃商品圖片編號',
  `RC_NO` int NOT NULL COMMENT '租賃商品類別編號',
  `RPI_IMG` mediumblob COMMENT '租賃圖片',
  PRIMARY KEY (`RPI_NO`),
  KEY `FK_RPI_RC_NO_idx` (`RC_NO`),
  CONSTRAINT `FK_RENTAL_PRODUCT_IMAGES_RC_NO` FOREIGN KEY (`RC_NO`) REFERENCES `RENTAL_CLASS` (`RC_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='租賃商品圖片';
