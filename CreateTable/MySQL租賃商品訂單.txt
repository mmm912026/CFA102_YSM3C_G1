CREATE TABLE `RENTAL_ORDER` (
  `RO_NO` int NOT NULL AUTO_INCREMENT COMMENT '租賃訂單編號',
  `MEM_NO` int NOT NULL COMMENT '會員編號',
  `RPL_NO` int NOT NULL COMMENT '租賃商品編號',
  `RO_STATUS` varchar(20) DEFAULT NULL COMMENT '訂單狀態',
  `RO_PAY_STATUS` varchar(20) DEFAULT NULL COMMENT '付款狀態',
  `RO_PAY_METHOD` varchar(20) DEFAULT NULL COMMENT '付款方式',
  `RO_SHIP_METHOD` varchar(20) DEFAULT NULL COMMENT '運送方式',
  `RO_SHIP_STATUS` varchar(20) DEFAULT NULL COMMENT '出貨狀態',
  `RO_SHIP_ADDRS` varchar(100) DEFAULT NULL COMMENT '配送地址',
  `RO_STARTTIME` datetime DEFAULT NULL COMMENT '租賃開始時間',
  `RO_ENDTIME` datetime DEFAULT NULL COMMENT '租賃結束時間',
  `RO_ONCERENTENDTIME` datetime DEFAULT NULL COMMENT '續租結束時間',
  `RO_RETURN_DATE` datetime DEFAULT NULL COMMENT '實際歸還時間',
  `RO_DAY` int DEFAULT NULL COMMENT '租賃天數',
  `RO_PRICE` int DEFAULT NULL COMMENT '租賃價格/每天',
  `RO_TOTALPRICE` int DEFAULT NULL COMMENT '訂單總價格',
  `RO_DEPOSIT` int DEFAULT NULL COMMENT '押金',
  `RO_DEPOSIT_STATUS` varchar(20) DEFAULT NULL COMMENT '押金狀態',
  `RO_RETURN_STATUS` varchar(20) DEFAULT NULL COMMENT '歸還狀態',
  `RO_RETURN_METHOD` varchar(20) DEFAULT NULL COMMENT '歸還方式',
  `RO_PRODUCT_STATUS` varchar(20) DEFAULT NULL COMMENT '商品狀態',
  `RO_REPAIRCOST` int DEFAULT NULL COMMENT '毀損扣除額',
  `RO_DELAY_DAYS` int DEFAULT NULL COMMENT '逾期(提前)天數',
  `RO_RETURN_DEPOSIT` int DEFAULT NULL COMMENT '實際歸還押金',
  PRIMARY KEY (`RO_NO`),
  KEY `FK_RENTAL_ORDER_MEM_NO_idx` (`MEM_NO`),
  KEY `FK_RENTAL_ORDER_RPL_NO_idx` (`RPL_NO`),
  CONSTRAINT `FK_RENTAL_ORDER_MEM_NO` FOREIGN KEY (`MEM_NO`) REFERENCES `MEMBER` (`MEM_NO`),
  CONSTRAINT `FK_RENTAL_ORDER_RPL_NO` FOREIGN KEY (`RPL_NO`) REFERENCES `RENTAL_PRODUCT_LIST` (`RPL_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='租賃商品訂單';
