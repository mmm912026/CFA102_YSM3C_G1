CREATE TABLE `RENTAL_PRODUCT_LIST` (
  `RPL_NO` int NOT NULL AUTO_INCREMENT COMMENT '評價編號',
  `RC_NO` int NOT NULL COMMENT '租賃商品類別編號',
  `RPL_SerialNum` varchar(50) DEFAULT NULL COMMENT '租賃商品產品序號',
  `RPL_NOTE` varchar(1000) DEFAULT NULL COMMENT '租賃商品註解',
  `RPL_STATUS` varchar(20) DEFAULT NULL COMMENT '租賃商品狀態',
  `RPL_RENTCOUNT` int DEFAULT '0' COMMENT '租賃次數',
  `RPL_JOINTTIME` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '租賃商品登入日期',
  PRIMARY KEY (`RPL_NO`),
  KEY `FK_RENTAL_PRODUCT_LIST_RC_NO_idx` (`RC_NO`),
  CONSTRAINT `FK_RENTAL_PRODUCT_LIST_RC_NO` FOREIGN KEY (`RC_NO`) REFERENCES `RENTAL_CLASS` (`RC_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='租賃商品資訊';
