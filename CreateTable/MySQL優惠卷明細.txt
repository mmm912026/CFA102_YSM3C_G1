CREATE TABLE `COUPON_LIST` (
  `CI_NO` int NOT NULL COMMENT '優惠劵編號',
  `MEM_NO` int NOT NULL COMMENT '會員編號',
  `CL_STATUS` varchar(45) NOT NULL COMMENT '狀態',
  KEY `CI_NO_idx` (`CI_NO`),
  KEY `FK_MEM_NO_idx` (`MEM_NO`),
  CONSTRAINT `FK_COUPON_LIST_CI_NO` FOREIGN KEY (`CI_NO`) REFERENCES `COUPON_INFORMATION` (`CI_NO`),
  CONSTRAINT `FK_COUPON_LIST_MEM_NO` FOREIGN KEY (`MEM_NO`) REFERENCES `MEMBER` (`MEM_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='優惠劵明細';
