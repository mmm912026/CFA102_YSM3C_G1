CREATE TABLE `STAFF` (
  `STAFF_NO` int NOT NULL AUTO_INCREMENT COMMENT '員工編號',
  `STAFF_NAME` char(20) DEFAULT NULL COMMENT '員工姓名',
  `STAFF_GENDER` char(10) DEFAULT NULL COMMENT '員工性別',
  `STAFF_PHONE` int DEFAULT NULL COMMENT '員工電話',
  `STAFF_EMAIL` varchar(65) DEFAULT NULL COMMENT '員工信箱',
  `STAFF_ADDRESS` varchar(65) DEFAULT NULL COMMENT '員工地址',
  `STAFF_ACCOUNT` char(20) DEFAULT NULL COMMENT '員工帳號',
  `STAFF_PASSWORD` char(20) DEFAULT NULL COMMENT '員工密碼',
  `STAFF_STA` varchar(20) DEFAULT NULL COMMENT '員工狀態',
  PRIMARY KEY (`STAFF_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='員工';
